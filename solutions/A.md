# A: A+B (Trial Problem)

```cpp
#include "bits/stdc++.h"

using namespace std;

void solve() {
    int a,b;
    cin >> a >> b;
    cout << a+b << '\n';
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int t;
    cin >> t;
    for (int i=0;i<t;i++) {
        solve();
    }
}
```
