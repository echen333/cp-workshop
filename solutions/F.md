#include "bits/stdc++.h"

using namespace std;

void solve() {
    int n;
    cin >> n;

}
int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int t;
    cin >> t;
    for (int i=0;i<t;i++) {
        solve();
    }
}
